; Shoe Shelves Cab Parameters

FOR EACH CAB ASSEMBLY

IF SSH_ON = NULL THEN
SSH_ON<BOOL> = 0
SSH_ON<STYLE> = 1
SSH_ON<Desc> = 'Shoe Shelf  ?'
End If

If SSH_ON == 0 THEN
DELETE SSH_ANGLE
;DELETE SSH_ADJUSTABLE
DELETE SSPINNOTCH
Delete SSPINNOTCHFRONT
Delete SSPINNOTCHSPACE
Delete HeelStop
EXIT
END IF

IF SSH_ANGLE = NULL THEN
SSH_ANGLE<deg> = 15
SSH_ANGLE<STYLE> = 1
SSH_ANGLE<Desc> = 'Shoe Shelf  Angle'
End If

If HeelStop = NULL Then
HeelStop<BOOL> = 0
HeelStop<STYLE> = 1
HeelStop<DESC> = 'Shoe Shelf Heel Stop?'
End If

;IF SSH_ADJUSTABLE = NULL THEN
;SSH_ADJUSTABLE<BOOL> = 0
;SSH_ADJUSTABLE<STYLE> = 1
;SSH_ADJUSTABLE<Desc> = 'Shoe Shelf Adjustables?'
;End If

IF SSH_BACK_OUTSET = NULL THEN
SSH_BACK_OUTSET<MEAS> = 0
SSH_BACK_OUTSET<STYLE> = 1
SSH_BACK_OUTSET<Desc> = 'Shoe Shelf Back Outset?'
End If

If SSPINNOTCH = NULL Then
SSPINNOTCH<BOOL> = 0
SSPINNOTCH<STYLE> = 1
SSPINNOTCH<DESC> = 'Shoe Shelf Pin Notches?'
End If

If SSPINNOTCH == 1 Then

If SSPINNOTCHFRONT = NULL Then
SSPINNOTCHFRONT<Meas> = 2
SSPINNOTCHFRONT<STYLE> = 1
SSPINNOTCHFRONT<DESC> = 'Shoe Shelf Pin Offset'
End If

If SSPINNOTCHSPACE = NULL Then
SSPINNOTCHSPACE<Meas> = 10
SSPINNOTCHSPACE<STYLE> = 1
SSPINNOTCHSPACE<DESC> = 'Shoe Shelf Pin Horz Space'
End If

Else
Delete SSPINNOTCHFRONT
Delete SSPINNOTCHSPACE
End If
